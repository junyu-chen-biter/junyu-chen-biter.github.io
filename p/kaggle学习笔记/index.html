def get_k_fold_data(k, i, X, y):
    # ËøôË°å‰ª£Á†Å‰ΩøÁî®‰∫ÜÊñ≠Ë®ÄÔºàassertÔºâÊäÄÊúØ„ÄÇ
    # Êñ≠Ë®ÄÊòØ‰∏ÄÁßçË∞ÉËØïÂ∑•ÂÖ∑ÔºåÁî®‰∫éÁ°Æ‰øùÊüê‰∏™Êù°‰ª∂‰∏∫Áúü„ÄÇ
    # Â¶ÇÊûú k ‰∏çÂ§ß‰∫é 1ÔºåÁ®ãÂ∫è‰ºöÊäõÂá∫ AssertionError ÂºÇÂ∏∏Âπ∂ÁªàÊ≠¢„ÄÇ
    # Âõ†‰∏∫ k Êäò‰∫§ÂèâÈ™åËØÅË¶ÅÊ±ÇËá≥Â∞ëÂ∞ÜÊï∞ÊçÆÂàÜ‰∏∫‰∏§ÈÉ®ÂàÜÔºåÊâÄ‰ª• k ÂøÖÈ°ªÂ§ß‰∫é 1„ÄÇ
    assert k > 1

    # ËøôË°å‰ª£Á†ÅËÆ°ÁÆóÊØè‰∏ÄÊäòÁöÑÊ†∑Êú¨Êï∞Èáè„ÄÇ
    # X.shape[0] Ë°®Á§∫ÁâπÂæÅÁü©Èòµ X ÁöÑË°åÊï∞ÔºåÂç≥Ê†∑Êú¨ÁöÑÊÄªÊï∞„ÄÇ
    # ‰ΩøÁî®Êï¥Èô§ËøêÁÆóÁ¨¶ // Á°Æ‰øùÁªìÊûú‰∏∫Êï¥Êï∞ÔºåÂæóÂà∞ÊØè‰∏ÄÊäòÂ§ßËá¥Áõ∏Á≠âÁöÑÊ†∑Êú¨Êï∞Èáè„ÄÇ
    fold_size = X.shape[0] // k

    # ÂàùÂßãÂåñËÆ≠ÁªÉÈõÜÁöÑÁâπÂæÅÂíåÊ†áÁ≠æ‰∏∫ None„ÄÇ
    # Âú®ÂêéÁª≠Âæ™ÁéØ‰∏≠ÔºåÊàë‰ª¨‰ºöÈÄêÊ≠•ÊûÑÂª∫ËÆ≠ÁªÉÈõÜ„ÄÇ
    X_train, y_train = None, None

    # ËøôÊòØ‰∏Ä‰∏™ for Âæ™ÁéØÔºåÁî®‰∫éÈÅçÂéÜ k ‰∏™Êäò„ÄÇ
    # j ÊòØÂæ™ÁéØÂèòÈáèÔºå‰ªé 0 Âà∞ k - 1 ‰æùÊ¨°ÂèñÂÄº„ÄÇ

        # ËøôË°å‰ª£Á†ÅÂàõÂª∫‰∫Ü‰∏Ä‰∏™ÂàáÁâáÂØπË±° idx„ÄÇ

        # ÂÆÉÊåáÂÆö‰∫Ü‰ªé j * fold_size Âà∞ (j + 1) * fold_size ÁöÑÁ¥¢ÂºïËåÉÂõ¥„ÄÇ
        # Ëøô‰∏™Á¥¢ÂºïËåÉÂõ¥ÂØπÂ∫î‰∫ÜÁ¨¨ j ÊäòÁöÑÊï∞ÊçÆÂú®Êï¥‰∏™Êï∞ÊçÆÈõÜ‰∏≠ÁöÑ‰ΩçÁΩÆ„ÄÇ
        idx = slice(j * fold_size, (j + 1) * fold_size)

        # ËøôË°å‰ª£Á†ÅÊ†πÊçÆÂàáÁâáÂØπË±° idx ‰ªéÁâπÂæÅÁü©Èòµ X ÂíåÊ†áÁ≠æÂêëÈáè y ‰∏≠ÊèêÂèñÁ¨¨ j ÊäòÁöÑÊï∞ÊçÆ„ÄÇ
        # X[idx, :] Ë°®Á§∫ÊèêÂèñÁâπÂæÅÁü©Èòµ X ‰∏≠Á¥¢ÂºïËåÉÂõ¥‰∏∫ idx ÁöÑÊâÄÊúâË°åÔºåÊâÄÊúâÂàóÁöÑÊï∞ÊçÆ„ÄÇ

        # Â¶ÇÊûúÂΩìÂâçÊäòÁöÑÁ¥¢Âºï j Á≠â‰∫éÊåáÂÆöÁöÑÈ™åËØÅÊäòÁ¥¢Âºï iÔºåËØ¥ÊòéËøô‰∏ÄÊäòÂ∞Ü‰Ωú‰∏∫È™åËØÅÈõÜ„ÄÇ
        if j == i:
            # Â∞ÜÁ¨¨ j ÊäòÁöÑÊï∞ÊçÆ‰Ωú‰∏∫È™åËØÅÈõÜÁöÑÁâπÂæÅÂíåÊ†áÁ≠æ„ÄÇ
            X_valid, y_valid = X_part, y_part

        elif X_train is None:
            # Â∞ÜÁ¨¨ j ÊäòÁöÑÊï∞ÊçÆ‰Ωú‰∏∫ËÆ≠ÁªÉÈõÜÁöÑÁâπÂæÅÂíåÊ†áÁ≠æ„ÄÇ
            X_train, y_train = X_part, y_part
        # Âê¶ÂàôÔºåËØ¥ÊòéËÆ≠ÁªÉÈõÜÂ∑≤ÁªèÊúâÈÉ®ÂàÜÊï∞ÊçÆÔºåÈúÄË¶ÅÂ∞ÜÂΩìÂâçÊäòÁöÑÊï∞ÊçÆÊ∑ªÂä†Âà∞Â∑≤ÊúâÁöÑËÆ≠ÁªÉÈõÜ‰∏≠„ÄÇ
        else:
            # ËøôË°å‰ª£Á†Å‰ΩøÁî®‰∫Ü torch.cat ÂáΩÊï∞ÔºåÂÆÉÁî®‰∫éÂ∞ÜÂ§ö‰∏™Âº†ÈáèÊ≤øÁùÄÊåáÂÆöÁöÑÁª¥Â∫¶ÊãºÊé•Âú®‰∏ÄËµ∑„ÄÇ
            # [X_train, X_part] ÊòØË¶ÅÊãºÊé•ÁöÑÂº†ÈáèÂàóË°®Ôºå0 Ë°®Á§∫Ê≤øÁùÄÁ¨¨ 0 Áª¥ÔºàÊ†∑Êú¨Áª¥Â∫¶ÔºâËøõË°åÊãºÊé•„ÄÇ
            # ËøôÊ†∑Â∞±Â∞ÜÂΩìÂâçÊäòÁöÑÁâπÂæÅÊï∞ÊçÆÊ∑ªÂä†Âà∞‰∫ÜËÆ≠ÁªÉÈõÜÁöÑÁâπÂæÅÊï∞ÊçÆ‰∏≠„ÄÇ
            X_train = torch.cat([X_train, X_part], 0)

            y_train = torch.cat([y_train, y_part], 0)
